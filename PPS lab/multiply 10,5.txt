//*Program to multiply 10,5 without using * operator
#include<stdio.h>
int main()
{
	int n1,n2,result = 0;
	printf("Enter two numbers n1,n2:");
	scanf("%d%d",&n1,n2);
	while(n2!=0){
		result = result + n1;
		n2--;
		
	}
	printf("multiplication of two numbers is:%d",result);
	return 0;
}
