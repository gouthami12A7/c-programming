#include<stdio.h>
int main()
{
	int w,c,n,totalweight,totalcost;
	printf("enter items weight:");
	scanf("%d",&w);
	printf("\nenter items cost:");
	scanf("%d",&c);
	printf("\nenter items number of units purchased:");
	scanf("%d",&n);

	printf("totalweight=%d",totalweight);
	totalcost=n*c;

	return 0;
	
	
}
