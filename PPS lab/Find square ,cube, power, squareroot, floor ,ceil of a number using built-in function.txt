//Find square ,cube, power, squareroot, floor ,ceil of a number using built-in function
#include<stdio.h>
#include<math.h>
int main()
{
	float num,p,f,c;
	printf("enter number:");
	scanf("%f",&num);
	printf("\n the square of a number is : %f",num*num);
	printf("\n the cube of a number is : %f",num*num*num);
	p = pow(num, 1);
	printf("\nthe power of the number is : %f",p);
	printf("\nthe squareroot of given number is : %f",sqrt(num));
	f = floor(num);
	c = ceil(num);
	printf("\n floor value : %f \nceil value : %f\n",f,c);
	return 0;
}
