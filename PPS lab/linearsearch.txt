void printArray(int num[],int n)
{
	int i;
	for(i=0;i<n;i++)
	{
		printf("%5d",num[i]);
	}
}
int LinearSearch(int A[],int n,int key)
{
	int i;
	for(i=0;i<n;i++)
	{
		if(key==A[i])
		return i+1;
	}
	return -1;
}
