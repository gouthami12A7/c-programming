//*function fibonacci*//
#include<stdio.h>
int fib(int);
int main()
{
	int n=0,i,res;
	printf("\n\n enter n");
	scanf("%d",&n);
	for(i=1;i<n;i++)
	{
		res = fib(i);
		printf("%4d",res);
		
	}
	return 0;
}
int fib(int n)
{
	if(n==1)
	return 0;
	if(n==2)
	return 1;
	else return fib(n-1)+fib(n-2);
}
