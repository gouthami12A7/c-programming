//Write a C program, which takes two integer operands and one operator from the user, performs the operation and then prints the result.// 
#include<stdio.h>
int main()
{
	int a,b;
	char ch;
	printf("\nEnter operands:");
	scanf("%d %d",&a,&b);
	printf("\n enter operand\n");
	scanf("%c",&ch);
	switch(ch)
	{
		case '+':printf("%d+%d = %d",a,b, a+b);
	    	break;
		case '-':printf("%d-%d=%d",a,b, a-b);
	    	break;
		case '*':printf("%d*%d = %d",a,b, a*b);
		    break;
		case '/':printf("%d/%d=%d",a,b, a/b);
	    	break;
		case '%':printf("%d",a%b);
	    	break;
	}
	return 0;
}
