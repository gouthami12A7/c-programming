//Program to read multiple line strings
#include<stdio.h>
#include<string.h>
int main()
{
	char Mstring[10],ch1,ch2;
	int i;
	printf("\nEnter the multiple lines string:");
	i=0;
	ch1='#';
	while(ch1!='$')
	{
		ch1 = getchar();
		Mstring[i]=ch1;
		i++;
		
	}
	Mstring[--i]='\0';
	printf("\n\n the string is \n\n");
	puts(Mstring);
}
