//*finding factorial using recursion*//
#include<stdio.h>
int factorial(int n);
int main()
{
	int n,res;
	printf("\n\nEnter n:");
	scanf("%d",&n);
	res=factorial(n);
	printf("\n factorial of %d is %d",n,res);
}
int factorial(int x)
{
	if(x==1)
	return 1;
	else
	return x*factorial(x-1);
}
