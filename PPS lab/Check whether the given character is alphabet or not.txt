//Check whether the given character is alphabet or not
#include<stdio.h>
int main()
{
	char ch;
	printf("Enter a character:");
	scanf("%c",&ch);
	if((ch>='a'&&ch<='z')||(ch>='A'&&ch<='Z'))
	printf("the entered character is a alphabet");
	else
	printf("the entered character is not a alphabet");
	return 0;
}
